Bài 1: Tìm kiếm tuyến tính ⭐
Tìm kiếm tính toán tuyến tính trên mảng số nguyên và mảng chuỗi. So sánh số bước.

#include <stdio.h>
#include <string.h>

// 1. Tìm kiếm tuyến tính trên mảng số nguyên
int linearSearchInt(int arr[], int n, int x, int *steps) {
    *steps = 0;
    for (int i = 0; i < n; i++) {
        (*steps)++;
        if (arr[i] == x) {
            return i; // Trả về chỉ số phần tử tìm thấy
        }
    }
    return -1; // Không tìm thấy
}

// 2. Tìm kiếm tuyến tính trên mảng chuỗi
int linearSearchString(char arr[][50], int n, char x[], int *steps) {
    *steps = 0;
    for (int i = 0; i < n; i++) {
        (*steps)++;
        if (strcmp(arr[i], x) == 0) {
            return i;
        }
    }
    return -1;
}

Bài 2: Tìm kiếm nhị phân ⭐⭐
Cài đặt Tìm kiếm nhị phân lặp + đệ quy. Tìm vị trí đầu tiên và cuối cùng của bản sao phần tử.

1. Cài đặt Lặp và Đệ quy (Cơ bản)
// Cách 1: Sử dụng vòng lặp
int binarySearchIterative(int arr[], int n, int x) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == x) return mid;
        if (arr[mid] < x) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

// Cách 2: Sử dụng đệ quy
int binarySearchRecursive(int arr[], int left, int right, int x) {
    if (left > right) return -1;
    int mid = left + (right - left) / 2;
    if (arr[mid] == x) return mid;
    if (arr[mid] < x) return binarySearchRecursive(arr, mid + 1, right, x);
    return binarySearchRecursive(arr, left, mid - 1, x);
}
2. Tìm vị trí đầu tiên và cuối cùng
int findFirstOccurrence(int arr[], int n, int x) {
    int left = 0, right = n - 1, result = -1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == x) {
            result = mid;       // Ghi nhận vị trí tạm thời
            right = mid - 1;    // Tiếp tục lùng sục bên trái
        } else if (arr[mid] < x) left = mid + 1;
        else right = mid - 1;
    }
    return result;
}

int findLastOccurrence(int arr[], int n, int x) {
    int left = 0, right = n - 1, result = -1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == x) {
            result = mid;       // Ghi nhận vị trí tạm thời
            left = mid + 1;     // Tiếp tục lùng sục bên phải
        } else if (arr[mid] < x) left = mid + 1;
        else right = mid - 1;
    }
    return result;
}


Bài 3: So sánh hiệu năng ⭐⭐
Đo thời gian tìm kiếm với n = 10.000, 100.000, 1.000.000 phần tử. Vẽ so sánh bảng.

#include <time.h>

void comparePerformance() {
    // Giả lập mảng lớn n phần tử đã sắp xếp...
    int n = 100000; 
    int *arr = (int*)malloc(n * sizeof(int));
    for(int i=0; i<n; i++) arr[i] = i;

    int target = n - 1; // Tìm phần tử cuối cùng để thấy rõ sự chênh lệch (Worst case)
    int steps = 0;

    // Đo Linear Search
    clock_t start = clock();
    linearSearchInt(arr, n, target, &steps);
    clock_t end = clock();
    double time_linear = (double)(end - start) / CLOCKS_PER_SEC * 1000; // Đổi sang mili-giây (ms)

    // Đo Binary Search
    start = clock();
    binarySearchIterative(arr, n, target);
    end = clock();
    double time_binary = (double)(end - start) / CLOCKS_PER_SEC * 1000;

    // Vẽ bảng so sánh ra màn hình console
    printf("\n+------------+-----------------+-----------------+\n");
    printf("| Thuat toan | So buoc lap     | Thoi gian (ms)  |\n");
    printf("+------------+-----------------+-----------------+\n");
    printf("| Linear     | %-15d | %-15.4f |\n", steps, time_linear);
    printf("| Binary     | %-15d | %-15.4f |\n", (int)log2(n), time_binary);
    printf("+------------+-----------------+-----------------+\n");
    
    free(arr);
}


Bài 4: 🔥 Dự Án Mini — Công cụ tìm kiếm thông minh ⭐⭐⭐
Cảm hứng: Brute Force Search — thuật toán trực quan hóa

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define MAX_CONTACTS 50

// Cấu trúc dữ liệu liên hệ
typedef struct {
    char name[50];
    char phone[15];
} Contact;

// Hàm so sánh hai số điện thoại (dùng cho thuật toán sắp xếp qsort)
int compareByPhone(const void *a, const void *b) {
    return strcmp(((Contact*)a)->phone, ((Contact*)b)->phone);
}

// Hàm chuyển đổi chuỗi thành chữ thường để tìm kiếm không phân biệt chữ hoa/chữ thường
void toLowerCase(char dest[], const char src[]) {
    int i = 0;
    while (src[i] != '\0') {
        if (src[i] >= 'A' && src[i] <= 'Z') {
            dest[i] = src[i] + 32;
        } else {
            dest[i] = src[i];
        }
        i++;
    }
    dest[i] = '\0';
}

// Hàm thực hiện Tìm kiếm mờ (Tìm kiếm tuyến tính hỗ trợ chuỗi con)
void searchByName(Contact book[], int n, char query[]) {
    char lowerQuery[50], lowerName[50];
    toLowerCase(lowerQuery, query);

    int foundCount = 0;
    int steps = 0;

    // Bắt đầu đo thời gian
    clock_t start = clock();
    
    printf("\n-> Tìm thấy kết quả:\n");
    for (int i = 0; i < n; i++) {
        steps++; // Mỗi lần kiểm tra một phần tử là một bước
        toLowerCase(lowerName, book[i].name);
        
        // Kiểm tra xem chuỗi tìm kiếm có nằm trong tên liên hệ không
        if (strstr(lowerName, lowerQuery) != NULL) {
            foundCount++;
            printf("   %d. %-22s - %s\n", foundCount, book[i].name, book[i].phone);
        }
    }
    
    clock_t end = clock();
    double timeSpent = (double)(end - start) / CLOCKS_PER_SEC * 1000; // Đổi sang ms

    // Nếu không tìm thấy kết quả, đưa ra gợi ý top 3 tên đầu tiên
    if (foundCount == 0) {
        printf("   Không tìm thấy kết quả nào phù hợp với \"%s\"!\n", query);
        printf("\n Gợi ý 3 liên hệ gần giống (có sẵn):\n");
        for (int i = 0; i < 3 && i < n; i++) {
            printf("   * %-22s - %s\n", book[i].name, book[i].phone);
        }
    } else {
        printf("   (Đã so sánh %d/%d phần tử — %.3fms)\n", steps, n, timeSpent);
    }
}

// Hàm thực hiện Tìm kiếm theo Số điện thoại (Sắp xếp rồi Tìm kiếm nhị phân)
void searchByPhone(Contact book[], int n, char queryPhone[]) {
    // 1. Sắp xếp danh bạ theo Số điện thoại trước khi tìm nhị phân
    qsort(book, n, sizeof(Contact), compareByPhone);

    int left = 0, right = n - 1;
    int steps = 0;
    int foundIdx = -1;

    clock_t start = clock();
    while (left <= right) {
        steps++;
        int mid = left + (right - left) / 2;
        int cmp = strcmp(book[mid].phone, queryPhone);

        if (cmp == 0) {
            foundIdx = mid;
            break; // Tìm thấy
        } else if (cmp < 0) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    clock_t end = clock();
    double timeSpent = (double)(end - start) / CLOCKS_PER_SEC * 1000;

    if (foundIdx != -1) {
        printf("\n-> Tìm thấy 1 kết quả:\n");
        printf("   1. %-22s - %s\n", book[foundIdx].name, book[foundIdx].phone);
        printf("   (Tìm kiếm nhị phân: hoàn thành sau %d bước — %.3fms)\n", steps, timeSpent);
    } else {
        printf("\n Không tìm thấy số điện thoại %s trong danh bạ!\n", queryPhone);
    }
}
